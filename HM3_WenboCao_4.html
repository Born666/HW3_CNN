def to_one_hot(y, num_class=10):
  n = y.shape[0]
  init_mat = numpy.zeros((n,num_class))
  idx_row = 0
  for i in y:
    init_mat[idx_row,i]=1
    idx_row += 1
  return init_mat
y_train_vec = to_one_hot(y_train)
